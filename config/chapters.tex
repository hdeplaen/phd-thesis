% CHAPTER STYLES
\usepackage{titlesec}

\titleformat{\chapter}[display]
{\Large\GaramondCaption\bfseries\centering}{{\MakeUppercase\chaptertitlename\hspace*{.5em}\thechapter}}{0pt}{\Huge\centering}

\titleformat{\section}
{\GaramondCaption\Large\bfseries}{\thesection}{1em}{}

\titleformat{\subsection}
{\GaramondCaption\large\bfseries}{\thesubsection}{1em}{}

\titleformat{\subsubsection}
{\GaramondCaption\normalsize\bfseries}{\thesubsubsection}{1em}{}


% QUOTE
\makeatletter
\newenvironment{chapquote}[2][.55\textwidth]
  {\setlength{\@tempdima}{#1}%
   \def\chapquote@author{#2}%
   \parshape 1 \@tempdima \dimexpr\textwidth-\@tempdima-2em\relax%
   \GaramondSwash}
  {\par\normalfont\hfill---\ {\GaramondCaption\chapquote@author}\hspace*{2em}\par\bigskip}
\makeatother
